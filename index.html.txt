  function showPage(pageId) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.getElementById(pageId).classList.add('active');

    document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('active'));
    const icons = {
      home: 'fa-home',
      search: 'fa-search',
      upload: 'fa-plus',
      message: 'fa-comment-dots',
      profile: 'fa-user'
    };
    document.querySelectorAll('.nav-btn').forEach(btn => {
      if (btn.querySelector('i').classList.contains(icons[pageId])) {
        btn.classList.add('active');
      }
    });
  }

  function uploadFile() {
    const fileInput = document.getElementById('fileInput');
    const file = fileInput.files[0];

    if (!file) {
      document.getElementById('uploadStatus').innerText = "No file selected!";
      return;
    }

    const fileRef = storageRef.child('pins/' + file.name);
    const uploadTask = fileRef.put(file);

    uploadTask.on('state_changed',
      (snapshot) => {
        const progress = (snapshot.bytesTransferred / snapshot.totalBytes) * 100;
        document.getElementById('uploadStatus').innerText = 'Upload is ' + progress.toFixed(0) + '% done';
      },
      (error) => {
        document.getElementById('uploadStatus').innerText = 'Upload failed. Please try again.';
      },
      () => {
        document.getElementById('uploadStatus').innerText = 'Upload completed!';
      }
    );
  }
